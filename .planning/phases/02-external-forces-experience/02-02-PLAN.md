    ```ts
    function getWeekDates(birthDate: string, weekIdx: number): string[] {
      const birth = new Date(birthDate)
      return Array.from({ length: 7 }, (_, i) => {
        const d = new Date(birth)
        d.setDate(d.getDate() + weekIdx * 7 + i)
        const mm = String(d.getMonth() + 1).padStart(2, '0')
        const dd = String(d.getDate()).padStart(2, '0')
        const yyyy = d.getFullYear()
        return `${yyyy}-${mm}-${dd}`
      })
    }
    ```

    Wikipedia fetch per date — use en.wikipedia.org/api/rest_v1/feed/onthisday/events/MM/DD (NOT api.wikimedia.org). AbortController timeout: 5000ms. On any error (network, timeout, non-OK status), return []. Parse data.events[] mapping to { year: e.year, text: e.text, url: e.pages?.[0]?.content_urls?.desktop?.page }.

    In trigger():
    1. Compute 7 date strings
    2. For dates already in cache, skip fetch (use cached value directly)
    3. For uncached dates, fetch in parallel via Promise.all
    4. Cache each result (including [] for failures)
    5. Build pool = all 7 dates' events flattened
    6. If pool.length === 0 → setStatus('error'); return
    7. Otherwise shuffle pool (Fisher-Yates: for i from n-1 down to 1, swap arr[i] with arr[Math.floor(Math.random()*(i+1))])
    8. setEvents(shuffled), setCurrentIdx(0), setStatus('loaded')

    For the cache hit path (all 7 dates cached): skip fetch entirely, build pool from cache, shuffle fresh (new shuffle each trigger), proceed to step 6.

    next() function: setCurrentIdx(i => (i + 1) % events.length)

    Fisher-Yates shuffle (do not use sort(() => Math.random() - 0.5)):
    ```ts
    function shuffle<T>(arr: T[]): T[] {
      const a = [...arr]
      for (let i = a.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1))
        ;[a[i], a[j]] = [a[j], a[i]]
      }
      return a
    }
    ```
